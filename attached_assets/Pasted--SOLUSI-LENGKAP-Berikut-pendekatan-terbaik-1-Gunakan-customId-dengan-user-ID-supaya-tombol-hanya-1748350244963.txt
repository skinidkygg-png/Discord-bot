async function handleGuideInteraction(interaction) {
  if (!interaction.isButton()) return;

  try {
    const [action, direction, userId] = interaction.customId.split('_');

    if (interaction.user.id !== userId) {
      return await interaction.reply({
        content: '‚ùå Kamu tidak bisa menggunakan tombol ini.',
        ephemeral: true
      });
    }

    // Cegah interaksi kedaluwarsa
    if (interaction.replied || interaction.deferred) return;

    // Dapatkan data halaman baru (next atau previous)
    const pageIndex = getPageIndex(interaction.message.embeds[0].title); // contoh fungsi kamu
    const newPageIndex = direction === 'next' ? pageIndex + 1 : pageIndex - 1;

    // Update embed + tombol baru (disable tombol lama secara otomatis karena ID beda)
    await interaction.update({
      embeds: [embedList[newPageIndex]],
      components: [getNavigationButtons(newPageIndex, userId)],
    });

  } catch (err) {
    console.error('‚ùå Error handling guide interaction:', err.message);
    // Optional: tanggapi interaksi kalau belum acknowledged
    if (!interaction.replied && !interaction.deferred) {
      await interaction.reply({ content: '‚ùå Terjadi kesalahan.', ephemeral: true });
    }
  }
}
3. Tombol hanya aktif sekali
Dengan cara di atas:

Karena customId mengandung user ID, hanya user itu yang bisa tekan

Setelah ditekan, interaction.update() mengganti tombol dengan customId baru (halaman baru)

Jadi tombol sebelumnya sudah tidak aktif untuk ditekan lagi (Discord akan anggap "expired")
